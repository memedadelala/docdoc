# 首页接口

## 首页

#### method

-    get

#### uri

-    /index

#### args

-    page  不传为 1 , 从1开始

#### res

```json

{
    code:0, //0为成功 其他大于0的的数字为错误标识
    msg:'', // success 或者其他错误标识 为英文
    format_msg:'',//直接输出
    data:{
        top:{ //头图
            name:'', //名称 只有page为1的第一个图有
            pic_url:'' , //图片地址 非全地址，需要拼接
            project_id:1 , // 项目id 或者 广告板id
            design_pic_id:12, // 项目中当前图片的id
            type:1, //1为项目 2位广告板
        },
        list:[ // 首页的瀑布流图片 20个
            {
                pic_url:'' , //图片地址 非全地址，需要拼接
                project_id:1 , // 项目id
                label:'', //产品标签 ，没有就为空
                design_pic_id:12, // 项目中当前图片的id
            }
        ],
        space:[ // 间隔信息
            {
                name:'', //名称
                desc:'', //描述
                id:'', //关键id
                pic_url:'', // 图片url
                type: 1, // 间隔内容的类型 1为用户信息列表 2为项目产品图 3为企业cart图
                job_status: 0 // 0为不求职 1为求职

            }
        ],
        top_type: 1 , // 类型 1为用户类型  3为企业
        top_list:[ // 头部列表
            {
                name:'', // 名称
                id:'', // 关键id
                pic_url:'',
            },

        ],
        page: 1 //回传的数据
    },

}


```
